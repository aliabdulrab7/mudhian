(RETAIL_SALES_KARATRATE.KARAT_RATE / item_curr_rate)

(RETAIL_SALES_KARATRATE.KARAT_RATE * item_curr_rate)

(RETAIL_SALES_KARATRATE.KARAT_RATE / item_curr_rate)                   

else                   
(RETAIL_SALES_KARATRATE.KARAT_RATE * item_curr_rate)

(RETAIL_SALES_KARATRATE.KARAT_RATE / item_curr_rate)

(RETAIL_SALES_KARATRATE.KARAT_RATE * item_curr_rate)

inner join metal_stock_master m on m.STOCK_CODE=MainStockCode                   

where R.MID=@intmidno AND ISNULL(gstvatonmaking,0)=1                   

select @SALESNONTAXABLEAMOUNT=ISNULL(@SALESNONTAXABLEAMOUNT,0)+ISNULL(SUM(CASE WHEN ISNULL(ExcludeGSTVAT,0)=1                   

then isnull(NETVALUEFC,0)                   

else 0 end) ,0)                   

from RETAIL_SALES_DETAIL RS INNER JOIN RETAIL_SALES R ON R.MID=RS.UNIQUEID                   

inner join diamond_stock_master d on d.STOCK_CODE=rs.STOCK_CODE                   

where R.MID=@intmidno AND ExcludeGSTVAT=1                   

SELECT @SRETURNNONTAXABLEAMOUNT= isnull(SUM(NETVALUEFC),0) FROM                   

RETAIL_SRETURN_DETAIL RS INNER JOIN RETAIL_SALES R ON R.SRETURNMID=RS.UNIQUEID                   

where R.MID=@intmidno and VAT_AMOUNTFC = 0                   

SELECT @SRETURNNONTAXABLEAMOUNT= ISNULL(@SRETURNNONTAXABLEAMOUNT,0) + isnull(SUM(METALVALUEFC),0) FROM                   

RETAIL_SRETURN_DETAIL RS INNER JOIN RETAIL_SALES R ON R.SRETURNMID=RS.UNIQUEID                   

INNER JOIN metal_stock_master m on m.STOCK_CODE=MainStockCode                   

WHERE R.MID=@intmidno and VAT_AMOUNTFC <> 0 and ISNULL(gstvatonmaking,0)=1                   

SELECT @SRETURNTAXABLEMOUNT=isnull(SUM(NETVALUEFC),0)                   

FROM RETAIL_SRETURN_DETAIL RS INNER JOIN RETAIL_SALES R ON R.SRETURNMID=RS.UNIQUEID                   

WHERE R.MID=@intmidno                   

SET @SRETURNTAXABLEMOUNT = ABS(ISNULL(@SRETURNNONTAXABLEAMOUNT,0) - ISNULL(@SRETURNTAXABLEMOUNT,0) )                   

SELECT @ADVTAX= isnull(SUM(VATAMTFC),0) FROM RETAIL_RECEIPT RR INNER JOIN RETAIL_SALES RS                   

ON RS.MID=RR.REFMID AND RS.VOCTYPE=RR.VOCTYPE                   

WHERE RS.MID=@intmidno                   

SELECT @OLDGOLDVALUE=-1*ISNULL(PURCHASE_VALUE_FC,0) FROM RETAIL_SALES WHERE MID=@intmidno                   

SELECT @SALESTAXAMOUNT=ISNULL(TOTALVAT_AMOUNTFC,0) FROM RETAIL_SALES WHERE MID=@intmidno                   

SELECT @SRETUNTAXAMOUNT=ISNULL(RS.TOTALVAT_AMOUNTFC,0) FROM RETAIL_SRETURN RS INNER JOIN RETAIL_SALES R                   

ON SRETURNMID=RS.MID WHERE R.MID=@intmidno                   

SELECT @SALESRETUNAMOUNT= -1*(SRETURN_VALUE_FC - RS.TOTALVAT_AMOUNTFC) FROM RETAIL_SRETURN RS INNER JOIN RETAIL_SALES R                   

ON SRETURNMID=RS.MID WHERE R.MID=@intmidno                   

--------------FINAL VALUE------------                   

SELECT @TAXABLEMOUNT=ISNULL(@SALESTAXABLEAMOUNT,0)-ISNULL(@SRETURNTAXABLEMOUNT,0)                   

SELECT @NONTAXABLEAMOUNT = ISNULL(@SALESNONTAXABLEAMOUNT,0) - ISNULL(@SRETURNNONTAXABLEAMOUNT,0) + ISNULL(@OLDGOLDVALUE,0)                   

SELECT @TAXAMOUNT=ISNULL(@SALESTAXAMOUNT,0)- ISNULL(@SRETUNTAXAMOUNT,0)
